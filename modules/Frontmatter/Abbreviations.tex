\abbr{BCI}{Brain-computer interface}{See \cref{sec:BCI}.}
\abbr{CA1}{``Cornu Ammonis'', subregion 1}{Area in the hippocampus where voltages are recorded from (see \cref{fig:brain-anatomy}).}
\abbr{CA3}{``Cornu Ammonis'', subregion 3}{Area in the hippocampus (see \cref{fig:brain-anatomy}). CA3 sends many axons (called ``Schafer collaterals'') to CA1.}
\abbr{ch.}{Channel}{}
\abbr{CSD}{Current source density}{The rate at which the positive charge surplus of a small volume decreases, $-\pdv*{\rho}{t}$. By charge conservation, equal to the net rate $\div{\J}$ at which positive charges flow out of the small volume. See \cref{sec:CSD}.}
\abbr{FIR}{Finite impulse response}{A linear filter whose output is a convolution of the input signal with some kernel.}
\abbr{GEVal}{Generalized eigenvalue}{See \cref{sec:generalized-eigenproblem}.}
\abbr{GEVec}{Generalized eigenvector}{See \cref{sec:generalized-eigenproblem}.}
\abbr{GRU}{Gated recurrent unit}{See \cref{sec:GRU_eqs}.}
\abbr{IIR}{Infinite impulse response}{A filter whose output at each timestep is a linear combination of both input samples and previous output samples.}
\abbr{IQR}{Interquartile range}{A measure of the spread of a set of one-dimensional values, that is robust to outliers. Difference between the 75th and the 25th data percentile.}
\abbr{KDE}{Kernel density estimate}{}
\abbr{LFP}{Local field potential}{The extracellular electric potential (see \cref{ch:LFP}).}
\abbr{LTP}{Long-term potentiation}{See \cref{sec:synaptic-plasticity}.}
\abbr{RMS}{Root-mean-square}{$\sqrt{\ev{x_t^2}}$ for a signal $x_t$.}
\abbr{RNN}{Recurrent neural network}{See \cref{ch:RNN}.}
\abbr{SNR}{Signal-to-noise ratio}{See \cref{sec:LSM}.}
\abbr{SOTA}{State of the art}{The algorithm currently used for SWR detection, namely an online single channel band-pass filter.}
\abbr{STDP}{Spike-timing-dependent plasticity}{See \cref{sec:synaptic-plasticity}.}
\abbr{SWR}{Sharp wave-ripple}{The pattern in the LFP that we want to detect in real-time. See \cref{ch:SWR}.}
\abbr{VHC}{Ventral hippocampal commisure}{See \cref{sec:disruption}.}
